### docker runコマンドでRobot Frameworkのテストケースを実行

docker run -it --name robot-test --rm \
-v "$HOME/.ssh/id_rsa:/tmp/id_rsa" \
-v "$PWD:/work:Z" \
-w /work \
--network="host" \
robopabot:1.0 \
bash -c "pabot \
-d output --timestampoutputs \
test01.robot"



### aliasに登録

alias pabot=\
"docker run --name robot-test --rm \
-v \"$HOME/.ssh/id_rsa:/tmp/id_rsa\" \
-v \"$PWD:/work:Z\" \
-w /work \
--network=\"host\" \
--user $(id -u):$(id -g) \
robopabot:1.0 \
pabot \
-d output --timestampoutputs"
